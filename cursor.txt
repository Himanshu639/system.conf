create table Stud_Marks(name varchar2(20), total_marks number);
create table Result(Roll number,Name varchar2(20), Class varchar2(10));
Declare
    mark1 number;
	class varchar2(20);
	name1 varchar2(20);
	roll1 number;
	i number;
Begin
for i in 1..3 loop
    name1:=:name1;
    mark1:=:mark1;
	roll1:=:roll1;
DBMS_OUTPUT.PUT_LINE("Name" || name1);
DBMS_OUTPUT.PUT_LINE("Marks" || mark1);
DBMS_OUTPUT.PUT_LINE("roll no." || roll1);
insert into Stud_Marks values(name1,mark1);
if mark1 between 1500 and 900 then
    class:='Distinction';
 ELSIF(mark1 BETWEEN 900 AND 989)
  THEN
   class:='first class';
  ELSIF(mark1 BETWEEN 825 AND 899)
  THEN
   class:='second class';
end if;
	DBMS_OUTPUT.PUT_LINE("Class" || class);
	insert into Result values(roll1,name1,class);
end loop;
commit;
Exception
    when others then
    	DBMS_OUTPUT.PUT_LINE('Error:'||SQLERRM);
END;
